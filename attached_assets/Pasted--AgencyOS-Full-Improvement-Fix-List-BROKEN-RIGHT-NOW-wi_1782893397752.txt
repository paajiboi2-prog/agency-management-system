What needs building:

New leave_balances table:
CREATE TABLE leave_balances (
  id text PRIMARY KEY,
  user_id text REFERENCES users(id) ON DELETE CASCADE,
  year integer NOT NULL,
  casual_total integer DEFAULT 12,
